Código do Web App (Apps Script)
function doGet(e) {
  return doPost(e);
}

function doPost(e) {
  var ss = SpreadsheetApp.openById(
    '1nL694BR_tkO5iHYHMoTpIelyMqXtktjIa87mWFeGmug'
  );
  var p = e ? e.parameter : {};
  var action = p.action || 'ping';
  var result = {};

  if (action === 'ping') {
    result = { ok: true };

  } else if (action === 'salvar') {
    var lanc = ss.getSheets()[1];
    var svcs = (p.svcs || '').split('|');
    var vals = (p.vals || '').split('|');
    var agora = new Date();
    for (var i = 0; i < svcs.length; i++) {
      if (!svcs[i]) continue;
      var dados = lanc.getRange('C10:C2000').getValues();
      var linha = 10;
      for (var j = 0; j < dados.length; j++) {
        if (dados[j][0] !== '') linha = 10 + j + 1;
        else break;
      }
      lanc.getRange(linha,3).setValue(svcs[i]);
      lanc.getRange(linha,4).setValue(Utilities.formatDate(
        agora,'America/Sao_Paulo','dd/MM/yyyy'));
      lanc.getRange(linha,5).setValue(Utilities.formatDate(
        agora,'America/Sao_Paulo','HH:mm'));
      lanc.getRange(linha,6).setValue(1);
      lanc.getRange(linha,7).setValue(parseFloat(vals[i]||0));
    }
    result = { ok: true, saved: svcs.length };

  } else if (action === 'listar') {
    var lanc = ss.getSheets()[1];
    var rows = lanc.getRange('C10:G500').getValues();
    var items = [];
    for (var i = 0; i < rows.length; i++) {
      if (!rows[i][0]) break;
      items.push({
        svc: rows[i][0],
        data: rows[i][1],
        hora: rows[i][2],
        qtd: rows[i][3],
        val: rows[i][4]
      });
    }
    result = { ok: true, items: items.reverse().slice(0,50) };
  }

  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}
